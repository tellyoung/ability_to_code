### 二分查找

```python
# 1. 有序数组
def base_search(nums, target):
    '''
    实现一个基础二分查找
    输入: 一个顺序list
    输出: 待查找的元素的位置
    '''
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = left + (right - left) // 2  # (left + right) >> 1 除2 向下取整
        if nums[mid] == target:
        	return mid
        elif target < nums[mid]:
        	right = mid - 1
        elif target > nums[mid]:
        	left = mid + 1
    return -1


def left_search(nums, target):
    '''
    存在多个相同数字
    左边界
    '''
	left, right = 0, len(nums) - 1
	while left <= right:
		mid = (left + right) // 2
        if nums[mid] == target:
        	right = mid - 1
        elif nums[mid] < target:
        	left = mid + 1
        elif nums[mid] > target:
        	right = mid - 1
	if left == len(nums) or nums[left] != target:
		return -1
    return left

def right_search(nums, target):
    '''
    右边界
    '''
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target:
        	left = mid + 1
        elif nums[mid] < target:
        	left = mid + 1
        elif nums[mid] > target:
        	right = mid - 1
	if right < 0 or nums[right] != target:
		return -1
    return right


# 因为我们初始化 right = len(nums) - 1
# 所以决定了我们的「搜索区间」是 [left, right]
# 所以决定了 while (left <= right)
# 同时也决定了 left = mid + 1 和 right = mid - 1


# 2. 旋转有序数组
def search(nums, target):
	'''
		在数组中搜索 target 值
		[2,5,6,0,0,1,2]
	'''
    left, right = 0, len(nums) - 1
    while left <= right:
        mid = (left + right) // 2
        if nums[mid] == target: return True
        if nums[mid] < nums[right]: # mid 在右半部分
            if target > nums[mid] and target <= nums[right]: # target 在右半部分
            	left = mid + 1
            else: # target 在左半部分
            	right = mid - 1

        elif nums[mid] > nums[right]:
            if target < nums[mid] and target >= nums[left]:
            	right = mid - 1
            else:
            	left = mid + 1

        elif nums[mid] == nums[right]: # 无法判断 存在相同元素
        	right -= 1 

    return False
```

##### 快速排序

```python
def quick_sort(arr, start, end):
    """
    快速排序
    """
    if start >= end:
        return
    base = arr[start]  # 设定基准元素
    low, high = start, end
    while low < high:
        while low < high and arr[high] >= base: # arr[high] 比基准元素小则跳出
            high -= 1
        arr[low] = arr[high]
        
        # 移动low找到符合条件的元素放在 high 处
        while low < high and arr[low] < base: # low指向的元素比基准元素小，则 low 向右移动
            low += 1
        arr[high] = arr[low]

    # low 与 high 重合 退出循环
    arr[low] = base  # 将基准元素放到 low 位置,左边的元素都比基准元素小,右边的元素都比基准元素大
    quick_sort(arr, start, low - 1)
    quick_sort(arr, low + 1, end)
```

```python
def merge(left, right):
    """
    将两个列表left, right按顺序融合为一个列表res
    """
    res = []
    i, j = 0, 0       # i和j是位置指针，i指left，j指right
    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            res.append(left[i])
            i += 1
        else:
            res.append(right[j])
            j += 1

    if i == len(left):
        for x in right[j:]:
            res.append(x)
    else:
        for x in left[i:]:
            res.append(x)

    return res

def merge_sort(arr):
    """
    归并排序
    """
    if len(arr) <= 1: return arr   # 只有一个元素 不用排序
    mid = len(arr) // 2

    left = merge_sort(arr[: mid])  # 子序列递归调用排序
    right = merge_sort(arr[mid:])
    return merge(left, right)
```

```python
def heapsort(arr):
    """
    堆排序
    """
    h = []
    for value in arr:
        heappush(h, value)
    return [heappop(h) for i in range(len(h))]
```

##### 冒泡排序
```python
def bubbling(array):
    """
    冒泡排序 从头到尾两两交换最大值放到最后
    """
    n = len(array)
    for i in range(0, n - 1):
        for j in range(1, n - i):
            if array[j] < array[j - 1]:
                array[j], array[j - 1] = array[j - 1], array[j]
    return array
```

##### 插入排序
```python
def insert(array):
    """
    插入排序 无序区每个元素插入有序区
    """
    n = len(array)
    for i in range(1, n):
        for j in range(i - 1, -1, -1):
            if array[j] <= array[j + 1]:
                break
            else:
                array[j], array[j + 1] = array[j + 1], array[j]
    return array
```

##### 选择排序
```python
def selection(array):
    """
    选择排序 循环选择当前最小元素放入数组头
    """
    n = len(array)
    for i in range(0, n - 1):
        min_id = i
        for j in range(i, n):
            if array[min_id] > array[j]:
                min_id = j
        array[i], array[min_id] = array[min_id], array[i]
    return array
```
